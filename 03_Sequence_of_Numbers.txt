# Create a sequence of numbers from 20 to 50
sequence1 <- 20:50

# Calculate the mean of numbers from 20 to 60
mean1 <- mean(20:60)

# Create a sequence of numbers from 51 to 91
sequence2 <- 51:91

# Calculate the sum of numbers from 51 to 91
sum2 <- sum(51:91)

# Print the results
cat("Sequence from 20 to 50: ", sequence1, "\n")
cat("Mean of numbers from 20 to 60: ", mean1, "\n")
cat("Sequence from 51 to 91: ", sequence2, "\n")
cat("Sum of numbers from 51 to 91: ", sum2, "\n")
